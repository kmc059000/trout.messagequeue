Queue Emails


MailMessage mailMessage = new MailMessage();

TroutFacade trout = new TroutFacade();
trout.MailMessageQueue.EnqueueMessage(mailMessage);

Send Emails


trout.MailMessageDequeuer.SendQueuedMessages();
